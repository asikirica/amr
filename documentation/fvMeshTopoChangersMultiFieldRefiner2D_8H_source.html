24Class
25 Foam::fvMeshTopoChangers::multiFieldRefiner2D
26
27Description
28 Dynamic mesh refinement/unrefinement based on volScalarField values.
29
30 Refinement can optionally be specified in a cellZone or in multiple
31 regions, each controlled by a different volScalarField.
32
33 Refinement can be achieved by combining multiple fields using logical
34 operators 'AND', 'OR' and 'NOT'. Logical operators are applied in a
35 sequential manner.
36
37 Governing field can be specified using the operator 'default'.
38
39 Geometry based refinement can be specified using the geometry field.
40
41Usage
42 Example of single field based refinement in all cells:
43 \verbatim
44 topoChanger
45 {
46 type multiFieldRefiner2D;
47
48 libs ("libfvMeshTopoChangers2D.so");
49
50 // How often to refine
51 refineInterval 1;
52
53 // Fields to use for refinement
54 refinementFields
55 {
56 alpha.water
57 {
58 // Refine field in between lower..upper
59 lowerRefineLevel 0.001;
60 upperRefineLevel 0.999;
61 operation default;
62 }
63
64 p_rgh
65 {
66 // Refine field in between lower..upper
67 lowerRefineLevel 1e4;
68 upperRefineLevel 1e5;
69 operation AND;
70 }
71
72 geometry
73 {
74 // Refine based on geometry constraint
75 type geometry;
76 surface "surface.stl";
77 mode inside;
78 buffer 0.0;
79 operation AND;
80 }
81 }
82
83 // Have slower than 2:1 refinement
84 nBufferLayers 1;
85
86 // Refine cells only up to maxRefinement levels
87 maxRefinement 2;
88
89 // Stop refinement if maxCells reached
90 maxCells 200000;
91
92 // Flux field and corresponding velocity field. Fluxes on changed
93 // faces get recalculated by interpolating the velocity. Use 'none'
94 // on surfaceScalarFields that do not need to be reinterpolated.
95 correctFluxes
96 (
97 (phi none)
98 (nHatf none)
99 (rhoPhi none)
100 (alphaPhi.water none)
101 (ghf none)
102 );
103
104 // Write the refinement level as a volScalarField
105 dumpLevel true;
106 }
107 \endverbatim
